import React, { Component, type Node } from "react";
import ScrollContainer from "../../elements/ScrollContainer";
import ContentContainer from "../../elements/ContentContainer";
import Paragraph from "../../elements/Paragraph";
import Title from "../../elements/Title";
import TextInput from "../../elements/TextInput";
import { bind } from "decko";
import Box from "../../elements/Box.web";
import Spacing from "../../components/Spacing";
import { withTheme } from "styled-components";
import { type Theme } from "../../utilities/theme";
import Button from "../../elements/Button";
 
type Props = {
  theme: Theme,
};
 
type State = {
  input: string,
};
 
class NewTab extends Component<Props, State> {
  state = {
    input: "",
  };
 
  @bind
  onChange(event) {
    const input = event.target.value;
    this.setState({
      input,
    });
  }
 
  render(): Node {
    const { theme } = this.props;
 
    return (
      <ContentContainer backgroundColor={theme.background.primary}>
        <Box column alignCenter padding="20px" fullWidth>
          <Spacing size={40} />
          <Title type="BOLD" size="LARGE">
            Remix
          </Title>
          <Spacing size={20} />
 
          <TextInput
            value={this.state.input}
            placeholder="Search for users, groups, or websites"
            secure
            onChange={this.onChange}
          />
 
          <Spacing size={20} />
 
          <Box>
            <Button title="New Friend" size="MEDIUM" />
            <Spacing size={20} />
            <Button title="New Group" size="MEDIUM" />
          </Box>
        </Box>
      </ContentContainer>
    );
  }
}
 
export default withTheme(NewTab);
