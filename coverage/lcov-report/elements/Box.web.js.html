type pad = {
  x?: number,
  y?: number,
};
 
function determinePadding(
  padding: pad | string | void,
  key: string = "padding"
) {
  if (!padding) return "";
 
  if (typeof padding === "string") return `${key}: ${padding}`;
 
  const { x, y } = padding;
 
  if (x && y) {
    return `${key}: ${y}px ${x}px;`;
  } else if (x && !y) {
    return `${key}: 0 ${x}px;`;
  } else {
    return `${key}: ${y}px 0;`;
  }
}
